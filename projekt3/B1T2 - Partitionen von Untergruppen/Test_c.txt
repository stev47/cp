##### Finde die kleinste nicht-zyklische Gruppe, welche keine nicht-triviale Partition besitzt.

 LogTo("Ergebnis_Test_c.txt");

 L := [ ];
 MonitoredGroups := [ ];
 GroupsWithNonTrivialPartition := [ ];
 GroupSizeList := [ ];


###


 Print("Test c)\n");


###


 for i in [1..10] do

	L[i] := AllGroups(i);

	for j in L[i] do
	
		Add( MonitoredGroups,  StructureDescription(j) );

	od;

 od;

 Print ("\n Untersuchte Gruppen", MonitoredGroups, "\n");


###


 for i in [1..Length(L)] do

	for j in L[i] do

		if HasNonTrivialPartition(j) = false and IsCyclic(j) = false then

			Add(GroupsWithNonTrivialPartition, j);
	
		fi;

	od;

 od;

 #Print ("\nGroupsWithNonTrivialPartition ", GroupsWithNonTrivialPartition, "\n\n");


###

 for i in GroupsWithNonTrivialPartition do

	Add(GroupSizeList, NrTuples(i,1) );

 od;

 #Print ("GroupSizeList ", GroupSizeList, "\n\n");


###


 Print("\n\nErgebnis von Test c)\n"); 
 Print("Die kleinste nicht zyklische Gruppe, welche eine nicht-triviale Partition besitzt, ist: ", StructureDescription( GroupsWithNonTrivialPartition[ Position( GroupSizeList, Minimum( GroupSizeList ) ) ]), "\n" ) ;
 Print("Diese Gruppe hat ", Minimum(GroupSizeList), " Elemente.\n");

 
 LogTo( );
